  def add_node size
    (0...size+1).each do |x| 
      (0...size).each do |y| 
        if y%2 == 0 and x%2 == 0 then
          @nodes.push(TotalWeightNode.new((x*3)+3,(y*4)))
        elsif x%2 == 1 and y%2 == 1 then
          @nodes.push(TotalWeightNode.new(((x*3)-3), (y*4)))
        end
      end
    end
  end
