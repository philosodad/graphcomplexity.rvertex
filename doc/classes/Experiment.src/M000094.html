  def print_to_file
    File.open("experiment6_av.tab", 'w') {|x|
      @average.each_index do |k|
        s = String.new
        @average[k].each_index do |i|
          s << @average[k][i].to_s + ?\t
        end
        x.puts(s)
      end
    }
    File.open("experiment6_fr.tab", 'w') {|x|
      @frequency.each_index do |k|

        end

      end
    }
    File.open("experiment6_rn.tab", 'w') {|x|

        s = String.new

          s << @runs[k][i].to_s + ?\t
        end

      end
    }

        s = String.new

          s << @fails[k][i].to_s + ?\t
        end
        x.puts(s)
      end
    }
