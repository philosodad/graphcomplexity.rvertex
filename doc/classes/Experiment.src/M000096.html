  def experiment
    [50, 150].each do |i| 
      [1.5,3,5].each do |k|
        200.times do
          @rg = RandomGraph.new(i, (i*k).to_i)
          @ag = PCDAllSimulator.new(@rg)
          @fails += @ag.long_sim[2]
          @attempts += 1
        end
        @rg = nil
        @ag = nil
      end
    end
  end
