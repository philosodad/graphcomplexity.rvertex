  def experiment
    [20,40,80].each do |i| 
      [3, 7].each do |j|
        20.times do
          rg = RandomGraph.new(i, i*j)
          puts "graphmade"
          @match = MatchSimulator.new(rg)
          @pcd = PCDAllSimulator.new(rg)
          puts "sims made"
          [@match, @pcd].each{|k| k.set}
          puts "sims set"
          [@match, @pcd].each{|k| k.sim}
          puts "simmed"
          a = @match.get_on_weight
          b = @pcd.get_on_weight
          if a < b then 
            over = true
            20.times do
              m = MatchSimulator.new(rg)
              m.set
              m.sim
              n = m.get_on_weight
              if n >= b then over = false end
              break unless !over
            end
            @count += 1 if over
          end
        end
      end
    end
  end
