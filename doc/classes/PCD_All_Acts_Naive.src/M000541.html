  def recieve_status n
    @onlist.add(n) if n.on
    if @on == nil 
      if n.on
        while !@covers.nodes[@cur].ids.include?(n.id)
          @cur += 1
          @cur = @cur%@covers.nodes.length
        end
        l = @covers.nodes.select{|k| !k.ids.include?(@id)}
        if l.empty? then 
          @on = true
          @next = :decided
        end
      elsif !n.on
        @on = true if (@weight < n.weight or n.off?)
        @next = :decided
      end
    end
  end
