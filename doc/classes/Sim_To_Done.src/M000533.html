  def sim *args
    args.length == 0 ? m = 500 : m = args[0]
    g = 0
    f = 0
    w = 0
    until all_done or g > m
      @rg.nodes.each{|k| k.do_next}
      @rg.nodes.each{|k| k.send_status}
      g+=1
    end
    if g > m then f = 1 end
    w = @rg.get_on_weight
    if f == 1 then g = 0 end
    $stdout.flush
    puts "S_T_DONE! #{@id} g: #{g}"
    return w, g, f 
  end
