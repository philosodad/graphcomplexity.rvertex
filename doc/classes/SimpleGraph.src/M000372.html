  def reset
    @nodes.each do |k|
      k.on = nil
      k.set_next(:analyze)
      k.set_now(:analyze)
    end
    @nodes.each {|k| k.set_ons}
    @nodes.each do |k|
      if k.weight == 0 then 
        k.on = false
        k.set_next(:out_of_batt)
      end
    end
  end
