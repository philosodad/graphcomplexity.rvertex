  def do_next
    @now = @next
    case @now
    when :choose
      @next = choose_nodetype
    when :notleaf
      @next = :pause
    when :pause
      @next = :root
    when :root
      @next = :choose
    when :notroot
      @next = :leaf
    when :leaf
      @next = :wait
    when :wait
      @next = :choose
    when :decided
      check_finished
    when :finish
      check_redundant
    when :done
      @next = :done
    end
  end
