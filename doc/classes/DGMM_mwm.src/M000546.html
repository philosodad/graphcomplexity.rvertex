  def choose_role
    @invites = []
    if @on != nil then
      @next = :done
    elsif @edges.to_a.select{|k| k.weight == nil}.empty?
      @on = false
      @next = :done
    elsif @weight < @neighbors.collect{|k| k.weight}.min
      @next = :invite
    elsif rand(2) == 1 then
      @next = :invite
    else
      @next = :listen
    end
    return @next
  end
