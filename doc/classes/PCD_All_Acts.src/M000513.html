  def recieve_status n
    @onlist.add(n) if n.on
    if @on == nil
      if n.on
        if n.weight == @onlist.to_a.collect{|k| k.weight}.min
          @covers.nodes.each_index do |k|
            c = @covers.nodes[k].ids
            if c.include?(@id) and c.include?(n.id) then
              @cur = k
              break
            end
          end
        end
      elsif !n.on
        @next = :analyze
      end
    end
  end
