  def initialize g
    @nodes = []
    @edges = g.edges
    g.nodes.each{|k| @nodes.push(SetNodeObs.new(k))}
    @nodes.each do |k|
      @edges.each do |j|
        if j.include?(k.id)
          k.neighbors.concat(@nodes.select{|i| (j - Set[k.id]).include?(i.id)})
        end
      end
    end
  end
