  def long_sim
    puts 'running sim!'
    t = 0
    s = ""
    f = 0
    while @rg.coverable? do
      i = sim
      if i[1] > 500 or i[2] > 0 then f += 1 end
      if @rg.covered? == false then
        f += 1 if f == 0
        puts "#{self.rg.class} simmed but not covered!"
        break
      end
      s = s + "#{i}, "
      t += @rg.reduce_by_min
      @rg.reset
    end
#    puts "#{@id} t: #{t}"
    s = s + "#{t}"
    if f != 0 then t = 0 end
    return t, s, f
  end
