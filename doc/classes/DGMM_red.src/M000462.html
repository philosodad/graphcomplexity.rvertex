  def check_finished
    if @neighbors.select{|k| k.now == :decided or k.now == :finish}.length == @neighbors.length
      if @on
        if @neighbors.select{|k| !k.on}.empty?
          @redundant = true
        end
      end
      @next = :finish
    else
      @next = :decided
    end
  end
